target_sources(fairseq2_cpp
    PRIVATE
        fairseq2.cpp
        model_loader.cpp
)
add_library(unity_lib)

target_sources(unity_lib
    PRIVATE
        lib/unity_lib.h
        lib/unity_lib.cpp
)

    PRIVATE

        lib/unity_lib.h
